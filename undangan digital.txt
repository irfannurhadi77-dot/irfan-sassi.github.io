    <script>
        // Fungsi menyalin nomor rekening
        function copyToClipboard(id) {
            const text = document.getElementById(id).innerText;
            navigator.clipboard.writeText(text).then(() => {
                alert("Nomor rekening berhasil disalin!");
            }).catch(err => {
                console.error('Gagal menyalin text: ', err);
            });
        }

        // Simulasi pengiriman ucapan secara dinamis
        const rsvpForm = document.getElementById('rsvpForm');
        const boxUcapan = document.getElementById('boxUcapan');
        const countUcapan = document.getElementById('countUcapan');
        let totalUcapan = 1;

        rsvpForm.addEventListener('submit', function(e) {
            e.preventDefault();
            
            const nama = document.getElementById('formNama').value;
            const kehadiran = document.getElementById('formKehadiran').value;
            const ucapan = document.getElementById('formUcapan').value;

            // Template badge kehadiran
            const badgeClass = kehadiran === 'Hadir' ? 'bg-green-50 text-green-600' : 'bg-red-50 text-red-600';

            // Elemen ucapan baru
            const ucapanBaru = document.createElement('div');
            ucapanBaru.className = 'bg-stone-50/80 p-3 rounded-xl border border-stone-100 text-xs animate-fade-in';
            ucapanBaru.innerHTML = `
                <div class="flex justify-between items-center mb-1">
                    <span class="font-semibold text-gray-700">${nama}</span>
                    <span class="text-[10px] px-1.5 py-0.5 ${badgeClass} rounded-md">${kehadiran}</span>
                </div>
                <p class="text-gray-500 font-light">${ucapan}</p>
            `;

            // Masukkan di paling atas list ucapan
            boxUcapan.insertBefore(ucapanBaru, boxUcapan.firstChild);
            
            // Reset form & update counter
            totalUcapan++;
            countUcapan.innerText = totalUcapan;
            rsvpForm.reset();

            alert('Terima kasih atas ucapan dan konfirmasi Anda!');
        });
    </script>
